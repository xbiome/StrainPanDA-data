library(reshape2)
library(dplyr)
melt_kraken= function(kraken_longum,sample_M,sample_B,sample_4M,sample_12M){

  Blong_M <- kraken_longum[sample_M,]
  names(Blong_M)<-sample_M
  Blong_B <- kraken_longum[sample_B,]
  names(Blong_B)<-sample_B
  Blong_4M <- kraken_longum[sample_4M,]
  names(Blong_4M)<-sample_4M
  Blong_12M <- kraken_longum[sample_12M,]
  names(Blong_12M)<-sample_12M



  Blong_M_melt <- cbind(melt(Blong_M),"M")
  colnames(Blong_M_melt)[2]<-"time"
  Blong_B_melt <- cbind(melt(Blong_B),"B")
  colnames(Blong_B_melt)[2]<-"time"
  Blong_4M_melt <- cbind(melt(Blong_4M),"4M")
  colnames(Blong_4M_melt)[2]<-"time"
  Blong_12M_melt <- cbind(melt(Blong_12M),"12M")
  colnames(Blong_12M_melt)[2]<-"time"

  melt_bind <- rbind(Blong_M_melt,Blong_B_melt,Blong_4M_melt,Blong_12M_melt)
  melt_bind_log <- melt_bind 
  melt_bind_log$value <- log10(melt_bind_log$value)
  return(melt_bind_log)
}
