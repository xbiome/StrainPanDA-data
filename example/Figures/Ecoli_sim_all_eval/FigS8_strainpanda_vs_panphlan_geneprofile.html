library(reshape2)
library(dplyr)
fill_JD <- function(JD_matrx, merged_panprof_heatmap){
  library(vegan)
  for(CP_i in colnames(JD_matrx)){
    for(GT_i in rownames(JD_matrx)){
      JD_matrx[GT_i,CP_i]<-vegdist(t(merged_panprof_heatmap[,c(CP_i,GT_i)]), method = "jaccard",binary=TRUE)
    }
  }
  return(JD_matrx)
}
