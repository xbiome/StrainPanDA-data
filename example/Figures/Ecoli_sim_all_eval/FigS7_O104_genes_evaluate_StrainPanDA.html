profile_truth<-rowSums(O104_ref[, O104_str])/3

gene_profile_merged <- merge(gene_profile[1:4], O104_ref[,O104_str], by=0)
gene_profile_merged[is.na(gene_profile_merged)] <- 0
rownames(gene_profile_merged)<-gene_profile_merged[,1]
gene_profile_merged <- gene_profile_merged[,-1]
O104_rows <- which(gene_profile_merged$"GCF_002983645_errfree_r0"==1)
gene_profile_merged_O104 <- gene_profile_merged[O104_rows,]

Ogene_expr <- intersect(rownames(gene_profile_merged_O104),unlist(func_gene))
Ogene_expr_heatmap <- gene_profile_merged_O104[Ogene_expr,1:4]
GCF_ID <- as.character(anno_table[Ogene_expr,"GFF_locus_tag"])
for(i_ID in 1:length(GCF_ID)){
  GCF_ID[i_ID]<-strsplit(GCF_ID[i_ID],split='.',fixed=TRUE)[[1]][1]
}
product_name <- paste(anno_table[Ogene_expr,"Product"],GCF_ID,sep="|")
while(sum(duplicated(product_name))>0){
  product_name[duplicated(product_name)]<-paste(product_name[duplicated(product_name)]," ",sep="")
}
rownames(Ogene_expr_heatmap)<-product_name


library(pheatmap)
pheatmap(Ogene_expr_heatmap, height=length(product_name)*0.2,width=dim(Ogene_expr_heatmap)[2]*1.5)
pheatmap(Ogene_expr_heatmap,filename = "eval_results/FigS7_heatmap/StrainPanDA_O104_functional_gene_pangenome_heatmap.pdf", height=length(product_name)*0.2,width=dim(Ogene_expr_heatmap)[2]*1.5)
